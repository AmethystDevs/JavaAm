Quick Syntax Reference

JavaAm provides a simple syntax layer on top of Java with cleaner declarations and concise control flow.

Variables

var name = "Alice";
var age = 25;
var score: double = 95.5;
val MAX_SIZE: int = 100;

Methods

fn greet: void(name: String) {
    System.out.println("Hello, " + name);
}

fn add: int(a: int, b: int) {
    return a + b;
}

public fn multiply: int(a: int, b: int) {
    return a * b;
}

Classes

class Animal {
    fn speak: void {
        System.out.println("Some sound");
    }
}

class Dog extends Animal {
    fn speak: void {
        System.out.println("Woof!");
    }
}

class Impl implements Runnable {
    fn run: void {
        System.out.println("Running...");
    }
}

Control Flow

if (x > 10) {
    System.out.println("Greater");
} else {
    System.out.println("Smaller or equal");
}

for (var i = 0; i < 10; i++) {
    System.out.println(i);
}

while (count < 5) {
    count = count + 1;
}

for (item: list) {
    System.out.println(item);
}
