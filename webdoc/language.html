JavaAm Programming Language

JavaAm is a simplified Java programming language that compiles directly to Java. It reduces boilerplate and makes Java code more readable while maintaining full Java compatibility.

Features

  • Simplified Syntax: Cleaner, more readable code compared to Java
  • Type Inference: Automatic type detection when not explicitly specified
  • Shorter Method Syntax: Use fn instead of return types upfront
  • Modern Features: Lambda expressions, enhanced for-each loops
  • Minecraft Support: Built-in support for Fabric, NeoForge, Forge, Paper, Spigot, and Purpur
  • Full Java Interoperability: Compiles to valid Java code

Hello World

package com.example;

class HelloWorld {
    fn main: void {
        var message = "Hello, JavaAm!";
        System.out.println(message);
    }
}

Variables and Types

// Type inference
var name = "Alice";

// Explicit typing
var age: int = 25;

// Immutable variable
val pi: double = 3.14159;

Methods

class Calculator {
    // Method with return type
    fn add: int(a: int, b: int) {
        return a + b;
    }

    // Void method
    fn greet: void(name: String) {
        System.out.println("Hello, " + name);
    }

    // Access modifiers
    public fn multiply: int(a: int, b: int) {
        return a * b;
    }

    private fn helper: void {
        System.out.println("Helper method");
    }
}

Control Flow

// If statement
if (age > 18) {
    System.out.println("Adult");
} else {
    System.out.println("Minor");
}

// While loop
var i: int = 0;
while (i < 10) {
    System.out.println(i);
    i = i + 1;
}

// For-each loop
for (item: list) {
    System.out.println(item);
}

Classes

class Animal {
    fn speak: void {
        System.out.println("Some sound");
    }
}

class Dog extends Animal {
    fn speak: void {
        System.out.println("Woof!");
    }
}

class Main {
    fn main: void {
        var dog = new Dog();
        dog.speak();
    }
}
